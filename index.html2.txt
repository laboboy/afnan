    // System & Chart State
    let chartInstance = null;
    let selectedMonthIndex = 12; // Start default at 12 months

    // Gompertz calculations
    function calculateGompertz(t, A, B, C) {
      return A * Math.exp(-B * Math.exp(-C * t));
    }

    function calculateDerivative(t, A, B, C) {
      const W_t = calculateGompertz(t, A, B, C);
      return W_t * B * C * Math.exp(-C * t);
    }

    // Interactive Feed/Bio-Phase Diagnoser
    function getDiagnosticExplanation(month, weight, derivative, inflectionMonth) {
      const adg = (derivative * 1000) / 30; // convert kg/month to g/day
      
      let phase = "";
      let strategy = "";
      
      if (month <= 6) {
        phase = "Pre-Weaning & Sucking Stage";
        strategy = `The calf is young (Weight: ${weight.toFixed(1)} kg) and relies largely on maternal milk. The growth rate is rising fast at +${derivative.toFixed(1)} kg/month (${adg.toFixed(0)}g/day). Management advice: Maintain high-grade forage for the heifer to boost lactation. Provide calf starter concentrates to ease the transition into high carbohydrate digestion.`;
      } else if (month > 6 && month <= Math.round(inflectionMonth)) {
        phase = "Post-Weaning Rapid Acceleration Phase";
        strategy = `This is the most critical developmental phase! The absolute derivative is surging toward its peak (+${derivative.toFixed(1)} kg/month). Since feed conversion ratio (FCR) is at its biological optimum, it is highly profitable to supplement grazing with energy-dense palm kernel cake (PKC) by-product feeds to maximize meat marbling potential.`;
      } else if (month > Math.round(inflectionMonth) && month <= 18) {
        phase = "Post-Inflection Deceleration Phase";
        strategy = `The growth curve has passed its mathematical inflection point (Month ${Math.round(inflectionMonth)}). The derivative is slowing to +${derivative.toFixed(1)} kg/month. While the cattle are still gaining weight, their biological rate of protein synthesis is slowing. Adjust your dietary focus away from hyper-protein concentrates toward cost-effective high-fiber silage.`;
      } else if (month > 18 && month <= 26) {
        phase = "Pubescent & Skeletal Consolidation";
        strategy = `Cattle are reaching market age (Weight: ${weight.toFixed(1)} kg) with a tapering derivative of +${derivative.toFixed(1)} kg/month. Growth is now mostly fat deposition rather than muscle or skeleton. This is the optimal 'slaughter window' for Malaysian markets where meat tenderness is prized. Delaying sale past this point will severely lower your farm's profitability because you will feed more for less gain.`;
      } else {
        phase = "Plateau / Mature Maintenance";
        strategy = `The cattle are fully mature, nearing the genetic asymptotic boundary of ${weight.toFixed(0)} kg. Growth velocity has bottomed out (+${derivative.toFixed(1)} kg/month). Feeding beef animals in this bracket is highly unprofitable as almost all feed nutrients are used purely for cellular maintenance, not live weight gain. Convert to breeding stock or liquidate immediately.`;
      }

      return { phase, strategy };
    }

    // Refresh and draw Chart.js
    function updateSimulator() {
      // Inputs
      const A = parseFloat(document.getElementById('sliderA').value);
      const B = parseFloat(document.getElementById('sliderB').value);
      const C = parseFloat(document.getElementById('sliderC').value);

      // DOM Value updates
      document.getElementById('valA').innerText = A;
      document.getElementById('valB').innerText = B.toFixed(1);
      document.getElementById('valC').innerText = C.toFixed(3);

      // Generate Data points for 0 to 36 months
      const months = Array.from({length: 37}, (_, idx) => idx);
      const weights = months.map(m => calculateGompertz(m, A, B, C));
      const derivatives = months.map(m => calculateDerivative(m, A, B, C));

      // Calculate Point of Inflection month: t_inflection = ln(B) / C
      const inflectionMonth = Math.log(B) / C;

      // Update Chart
      const ctx = document.getElementById('growthChart').getContext('2d');
      
      if (chartInstance) {
        chartInstance.destroy();
      }

      chartInstance = new Chart(ctx, {
        type: 'line',

          labels: months,

              display: false // Handled customized below chart

                label: function(context) {
                  const val = context.raw.toFixed(2);
                  const label = context.dataset.label;
                  return `${label.split(' (')[0]}: ${val}`;
                }
              }
            }
          },
          onClick: (evt, activeElements) => {
            if (activeElements.length > 0) {

    // Update numbers & text in live calculations hub
    function updateCalculationHub(A, B, C, inflectionMonth) {
      const month = selectedMonthIndex;
      const weight = calculateGompertz(month, A, B, C);
      const derivative = calculateDerivative(month, A, B, C);
      const adg = (derivative * 1000) / 30; // standard bioscience conversion formula

      // Update numbers
      document.getElementById('selectedMonth').innerText = month;
      document.getElementById('liveWeightVal').innerText = weight.toFixed(1);
      document.getElementById('liveDerivativeVal').innerText = `+${derivative.toFixed(1)}`;
      document.getElementById('liveAdg').innerText = adg.toFixed(0);

      // Diagnostics & Strategy text
      const diagnostic = getDiagnosticExplanation(month, weight, derivative, inflectionMonth);
      document.getElementById('growthPhaseLabel').innerText = diagnostic.phase;
      document.getElementById('analysisExplanation').innerHTML = diagnostic.strategy;

      // Color code growth phase
      const phaseLabel = document.getElementById('growthPhaseLabel');
      if (month <= 6) {
        phaseLabel.className = "text-xl font-bold text-sky-400";
      } else if (month <= Math.round(inflectionMonth)) {
        phaseLabel.className = "text-xl font-bold text-emerald-400";
      } else if (month <= 18) {
        phaseLabel.className = "text-xl font-bold text-yellow-400";
      } else {
        phaseLabel.className = "text-xl font-bold text-amber-500";
      }
